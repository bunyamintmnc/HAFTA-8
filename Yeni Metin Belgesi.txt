SELECT last_name, category_name, first_name
FROM dbo.kategoriler
INNER JOIN dbo.musteriler
	ON dbo.kategoriler.category_id = dbo.musteriler.customer_id


SELECT product_id, product_name, brand_id, category_id, model_year, list_price
FROM	urunler
where (model_year between 2016 and 2018)
AND (list_price > 1000)
AND (category_id < 5)


SELECT COUNT(*) AS Expr1
FROM	urunler
where (model_year between 2016 and 2018) OR
		(list_price >
				(SELECT	AVG(list_price) AS Expr1
					FROM	urunler AS urunler_1))

SELECT customer_id, first_name, last_name, phone, email, street, city, state, zip_code
FROM musteriler
WHERE (first_name LIKE 'c%')

SELECT customer_id, first_name, last_name, phone, email, street, city, state, zip_code
FROM musteriler
WHERE (first_name LIKE 'b%n')


SELECT kategoriler.category_name
FROM urunler
INNER JOIN kategoriler ON kategoriler.category_id=urunler.category_id
where	(list_price >
				(SELECT	AVG(list_price) AS Expr1
					FROM	urunler AS urunler_1))

order by list_price
